
    Фоновый скрипт

    "background": {
      "scripts": [
        "js/background.js"
        // общий скрипт(фоновый скрипт),
        // который запускается при запуске браузера;
      ]
      //,
      //"page": "background.html"
      //Фоновая страница — это невидимая страница, которая нужна для того, чтобы объединять все вкладки.
      // Ссылки на ресурсы можно указывать как в файле манифеста,
      // так и на самой странице background.html
      // Расширению часто требуется общий для всех вкладок скрипт,
      // который обеспечивает их взаимодействие, хранит состояние.
      // В конце концов, объединяет в одном месте весь используемый код,
      // вместо того чтобы загружать его для каждой вкладки индивидуально.
      // Фоновая страница – это невидимая страница, которая как раз и содержит такой скрипт.
      // Их бывает два типа: persistent (постоянные) и event (управляемые событиями).
      // Persistent, как можно догадаться, загружена постоянно,
      // а event-страница загружается только при необходимости, когда возникает связанное с ней событие.
    },



    let level = 0;
    let langFrom = 'eng';
    let langTo = 'eng';
    let power = false;

    chrome.storage.sync.get('rangeInput', obj => level = obj.rangeInput);
    chrome.storage.sync.get('langFrom', obj => langFrom = obj.langFrom);
    chrome.storage.sync.get('langTo', obj => langTo = obj.langTo);


    chrome.runtime.onMessage.addListener(
        msgObj => {
            let params = JSON.parse(msgObj);
            if (params.power) {
                window.onload = function () {
                    translateThis(level, langFrom, langTo);
                }
            }
        });
